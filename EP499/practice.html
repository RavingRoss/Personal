def graphImpulse():
    t = np.linspace(0,10,1000)
    t1,y1 = co.impulse_response(G, t)
    plt.plot(t1,y1)
    plt.xlabel('time (s)')
    plt.ylabel('Amplitude')
    plt.grid()
    plt.show()
def graphstep():
    t = np.linspace(0,20,1000)
    t1,y1 = co.step_response(G, t)
    plt.plot(t1,y1)
    plt.xlabel('time (s)')
    plt.ylabel('Amplitude')
    plt.grid()
    plt.show()
def graphForced():
    t = np.linspace(0,30,1000)
    u = np.sin(t)
    t1,y1 = co.forced_response(G,t,u)
    plt.plot(t1,y1)
    plt.xlabel('time (s)')
    plt.ylabel('Amplitude')
    plt.grid()
    plt.show()
